# Force CMake version 3.1 or above
cmake_minimum_required (VERSION 3.1)

# This project has the name: 18-complex-conjugate
project (18-complex-conjugate)

# Create an executable named 'complex-conjugate' from the source file 'complex-conjugate.cxx'
add_executable(complex-conjugate src/complex-conjugate.cxx)

# Define C++11 standard
set_property(TARGET complex-conjugate PROPERTY CXX_STANDARD 11)

# Add test called "complex-conjugate-test" based on the executable "complex-conjugate"
add_test (complex-conjugate-test complex-conjugate)
